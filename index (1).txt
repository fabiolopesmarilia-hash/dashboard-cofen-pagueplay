    // Função para enviar para a nuvem
    function salvarEvento() {
        const txt = document.getElementById('evento').value;
        const hr = document.getElementById('hora').value;
        
        if(!txt || !hr) {
            alert("Fábio, preencha o texto e a hora antes de salvar!");
            return;
        }

        database.ref('agenda_cofen').push({
            texto: txt,
            hora: hr
        });

        document.getElementById('evento').value = "";
        document.getElementById('hora').value = "";
    }

    // Função que "escuta" a nuvem e atualiza a tela de todo mundo
    database.ref('agenda_cofen').on('value', (snapshot) => {
        const lista = document.getElementById('lista-eventos');
        lista.innerHTML = "";
        snapshot.forEach((item) => {
            const data = item.val();
            const li = document.createElement('div');
            li.className = 'item-agenda';
            li.innerHTML = `<span><strong>${data.hora}</strong> - ${data.texto}</span>`;
            lista.prepend(li); // O mais recente aparece no topo
        });
    });
