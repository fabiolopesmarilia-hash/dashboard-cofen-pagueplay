<script>
    function salvarAgenda() {
        const texto = document.getElementById('feedback-txt').value;
        const hora = document.getElementById('feedback-time').value;
        const lista = document.getElementById('lista-agenda');

        if(texto === "" || hora === "") { alert("Preencha o que e o horário!"); return; }

        const li = document.createElement('li');
        li.className = 'item-agenda';
        li.innerHTML = `<span><strong>${hora}</strong> - ${texto}</span> <button class="btn-del" onclick="this.parentElement.remove()">X</button>`;
        
        lista.appendChild(li);

        // Limpa campos
        document.getElementById('feedback-txt').value = "";
        document.getElementById('feedback-time').value = "";
    }
</script>
