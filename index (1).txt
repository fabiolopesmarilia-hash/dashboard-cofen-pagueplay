        <input type="text" id="txt" placeholder="Evento/Feedback" style="width:70%; padding:8px;">
        <input type="time" id="hora" style="width:25%; padding:8px;">
        <button class="btn-add" onclick="salvarNaNuvem()" style="margin-top:10px;">SALVAR PARA TODOS</button>

<script>
    // ESTA É A LÓGICA DE CONEXÃO - O provedor precisará colocar as chaves do Firebase aqui
    // Por enquanto, esta função simula a gravação. 
    // Para funcionar entre cidades, você precisa criar uma conta no firebase.google.com
    function salvarNaNuvem() {
        const texto = document.getElementById('txt').value;
        const hora = document.getElementById('hora').value;
        if(!texto || !hora) return;

        const li = document.createElement('li');
        li.className = 'item-agenda';
        li.innerHTML = `<span><strong>${hora}</strong> - ${texto}</span>`;
        document.getElementById('lista-nuvem').appendChild(li);
        
        alert("Salvo! (Nota: Para sincronizar entre cidades, conecte seu Firebase API Key no código)");
    }
</script>
