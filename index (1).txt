<script>
    // ESTA CONFIGURAÇÃO PRECISA SER A SUA (Crie no console.firebase.google.com)
    // Por enquanto, o código abaixo apenas simula localmente até você colocar sua API Key
    function salvarEvento() {
        const txt = document.getElementById('evento').value;
        const hr = document.getElementById('hora').value;
        if(!txt || !hr) return alert("Preencha tudo!");

        const lista = document.getElementById('lista-eventos');
        const li = document.createElement('li');
        li.className = 'item-agenda';
        li.innerHTML = `<span><strong>${hr}</strong> - ${txt}</span>`;
        lista.appendChild(li);
        
        alert("Agendado! Para a Brenda ver em outra cidade, precisamos configurar seu Firebase pessoal.");
    }
</script>
